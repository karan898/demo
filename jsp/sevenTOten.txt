// 1. AccountBean.java (Business Logic)

package bean;

import java.io.Serializable;

public class AccountBean implements Serializable {

private int accountNo;
private String name;
private double balance;

// Default Constructor
public AccountBean() {
balance = 0.0;
}

// Getters & Setters
public int getAccountNo() {
return accountNo;
}

public void setAccountNo(int accountNo) {
this.accountNo = accountNo;
}

public String getName() {
return name;
}

public void setName(String name) {
this.name = name;
}

public double getBalance() {
return balance;
}

// Business Methods
public void deposit(double amount) {
balance += amount;
}

public boolean withdraw(double amount) {
if (amount <= balance) {
balance -= amount;
return true;
}
return false;
}
}

 1. sort.tag (Custom Tag Logic)

<%@ tag body-content="empty" %>
<%@ attribute name="numbers" required="true" %>

<%
String[] arr = numbers.split(",");
int[] nums = new int[arr.length];

// Convert to int
for (int i = 0; i < arr.length; i++) {
nums[i] = Integer.parseInt(arr[i].trim());
}

// Sorting (Ascending)
for (int i = 0; i < nums.length; i++) {
for (int j = i + 1; j < nums.length; j++) {
if (nums[i] > nums[j]) {
int temp = nums[i];
nums[i] = nums[j];
nums[j] = temp;
}
}
}
%>

<h3>Sorted Numbers:</h3>

<%

out.print(n + " ");
}

 3. UserListener.java

package listener;

import javax.servlet.;
import javax.servlet.http.;

public class UserListener implements HttpSessionListener {

public void sessionCreated(HttpSessionEvent se) {

ServletContext context = se.getSession().getServletContext();

Integer count = (Integer) context.getAttribute("users");

if (count == null) count = 0;

count++;

context.setAttribute("users", count);

System.out.println("User Entered: " + count);
}

public void sessionDestroyed(HttpSessionEvent se) {

ServletContext context = se.getSession().getServletContext();

Integer count = (Integer) context.getAttribute("users");

if (count == null) count = 0;

count--;

context.setAttribute("users", count);

System.out.println("User Left: " + count);
}
}
